cmake_minimum_required (VERSION 2.6)
project (GLUTCraft)
set (PROJECT_DIR .)
set (PROJECT_BUILD_DIR "${PROJECT_DIR}/build")
set (EXECUTABLE_OUTPUT_PATH "${PROJECT_BUILD_DIR}/bin")
set (LIBRARY_OUTPUT_PATH "${PROJECT_BUILD_DIR}/lib")
set (PROJECT_SOURCE_DIR "${PROJECT_DIR}/src")
set (PROJECT_INCLUDE_DIR "${PROJECT_DIR}/include")
include_directories ("${PROJECT_INCLUDE_DIR}")
add_library (utils STATIC "${PROJECT_SOURCE_DIR}/utils.c")
add_executable (GLUTCraft "${PROJECT_SOURCE_DIR}/main.c")
if(WIN32)
message ("Platform: Windows")
target_link_libraries (GLUTCraft opengl32 glu32 freeglut utils)
endif(WIN32)
if(UNIX)
message ("Platform: UN*X")
target_link_libraries (GLUTCraft GL GLU glut utils)
endif(UNIX)
file (COPY "${PROJECT_INCLUDE_DIR}" DESTINATION "${PROJECT_BUILD_DIR}")
install (TARGETS GLUTCraft DESTINATION bin)
install (TARGETS utils DESTINATION lib)
install (FILES include/utils.h DESTINATION include)
